Integration Code 
import requests

files = {
    'file1': open('path/to/your/file1.txt', 'rb'),
    'file2': open('path/to/your/file2.txt', 'rb'),
    # Add more files as needed
}

data = {
    'workflowID': '{{ workflow.id }}',
}

response = requests.post(
    'https://vandalizer.nkn.uidaho.edu/workflow/run',
    headers={
        'x-api-key': '{{ current_user.id }}',
        'Accept': 'application/json',
    },
    data=data,
    files=files,
)

print(response.json())
