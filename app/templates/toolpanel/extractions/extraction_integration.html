Basic Setup

Start Extraction


import requests
import json

files = {
    'file': open('path/to/your/file1.pdf', 'rb'),
    'file': open('path/to/your/file2.pdf', 'rb'),
    # Add more files as needed
}

data = {
    'search_set_uuid': '{{ search_set.uuid }}',
}

response = requests.post(
    '{{ base_url }}/tasks/run_integrated',
    headers={
        'x-api-key': '{{ user.id }}',
        'Accept': 'application/json',
    },
    data=data,
    files=files,
)

result = response.json()
print(result)
